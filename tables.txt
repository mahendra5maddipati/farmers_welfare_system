#admin

CREATE TABLE ADMIN (
id INTEGER PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(55) NOT NULL,
email VARCHAR(255) NOT NULL UNIQUE,
password VARCHAR(50) NOT NULL,
created_on TIMESTAMP
);
